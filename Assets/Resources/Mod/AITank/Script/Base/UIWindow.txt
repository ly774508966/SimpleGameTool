local base = Class("UIWindow")

function base:ctor()
	self.prefabName = ""
	self.gameObject = nil
end

function base:init(func)
	if not self.gameObject then
		ResourceManager.GetInstance():SingleLineLoad(self.prefabName, function(obj)
			self.gameObject = GameObject.Instantiate(obj, Vector3.zero, Vector3.zero)
			self.transform = self.gameObject:GetComponent(RectTransform)
			
			if func then
				func()
			end
		end)
	else
		if func then
			func()
		end
	end
end

function base:show()
	self.gameObject:SetActive(true)
end

function base:hide()
	self.gameObject:SetActive(false)
end

function base:push(lastName)
	self.last = lastName
end

function base:pop()
	self.last = nil
end

return base
